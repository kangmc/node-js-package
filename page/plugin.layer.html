jQuery Plugin : layer

레이어 형태의 팝업을 띄운다.
기본적으로는 하나만 존재하도록 내부 로직이 처리되기 때문에 별도로 띄워야할 경우 id를 부여해서 구분해야함.
function: show, hide, remove, add, clear
최초 선언시 body에 자동 추가됨

샘플 바로가기

Global

동일하게 적용되어야하는 script는 scripts/jquery/jquery.init.js 파일에 작성한다.

Use Case

			(function( $, window, document){
				$(function(){
					
					
					
				});
			})( jQuery, window, document );

Page

page 에 작성되어야 하는 script 는 </body> Tag 바로 위에 작성한다.

Use Case

			<script>
			(function( $, window, document){
				$(function(){
					
					//======================== page initialize ===================
					var popup = $.layerPopup(
						id: 'layer의 최상위 element에 지정되는 id, 기본값은 layer'
						html: '팝업내용',
						target: '대상 element의 id',
						attr: {
							style: 'css style 지정',
							... : 추가 할 attribute는 attr내부에 작성 
						}
					);
					
					popup.show(); // 보이기
					popup.hide(); // 숨기기
					popup.remove(); // 제거
					//============================================================
					
				});
			})( jQuery, window, document );
			</script>
			</body>
