2. Validate

Validate 는 Element 의 Attribute 추가로 설정할 수 있다. 
			
			<!-- required : 필수 입력 항목 -->
			<input type="text" required="required" />
			
			<!-- email : 이메일 형식 (aaa@domain.com) -->
			<input type="text" email="email" />
			
			<!-- url : 웹페이지 주소 형식 (htttp://www.domain.com) -->
			<input type="text" url="url" />
			
			<!-- date : 날짜 형식 체크 -->
			<input type="text" date="date" />
			
			<!-- dateISO : ISO 표준 날짜 형식 체크 -->
			<input type="text" dateISO="dateISO" />
			
			<!-- number : 실수 형식 체크 (ex : 1, 1.3) -->
			<input type="text" number="number" />
			
			<!-- digits : 자연수 형식 체크 -->
			<input type="text" digits="digits" />
			
			<!-- creditcard : 카드번호 형식 체크 -->
			<input type="text" creditcard="creditcard" />
			
			<!-- engnum : 영문, 숫자 허용 -->
			<input type="text" engnum="engnum" />
			
			<!-- equalTo : 값 비교 id selector 지정 -->
			<input type="text" equalTo="#id" />
			
			<!-- maxlength : 최대 글자수 -->
			<input type="text" maxlength="3" />
			
			<!-- minlength : 최소 글자수 -->
			<input type="text" minlength="2" />
			
			<!-- length : 글자수 지정 -->
			<input type="text" length="3" />
			
			<!-- rangelength : 글자수 범위지정 -->
			<input type="text" rangelength="3,5" />
			
			<!-- range : 입력 숫자 범위 지정 -->
			<input type="text" range="3,5" />
			
			<!-- max : 입력 숫자 최대값 -->
			<input type="text" max="5" />
			
			<!-- min : 입력 숫자 최소값 -->
			<input type="text" min="2" />

3. 문구 우선 순위

1. 타겟 문구
a. 요소의 data-title Attribute > b. wrapping 하고 있는 label 요소의 문구 > c. 요소의 placeholder > d. 요소의 name

2. 설명 문구
요소의 data-message Attribute > 기본 문구
기본 문구 설정은 scripts/jquery/jquery.utils.js 파일의 validate.extend 함수에서 수정할 수 있다. 
			
			<input type="text" data-title="이름" />
			
			
			<label>이름
			<input type="text" data-title="" />
			</label>
			
			
			<input type="text" placeholder="이름" />
			
			
			<input type="text" name="name" />
			
			
			<input type="text" name="name" data-message="은 도메인 형식입니다." />

자동설정 제거 옵션

form Tag 에 novalidate="novalidate" Attribute 를 추가하여 validate를 예외로 처리할 수 있다. 
			<form novalidate="novalidate">
			...
			<form>

조건 추가

validate 조건을 추가할 수 있다.
참조 : jQuery.validator.addMethod

				return this.optional(element) || /^http:\/\/mycorporatedomain.com/.test(value);
			}, "Please specify the correct domain for your documents");
