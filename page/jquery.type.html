Type Check

jQuery.is... Method 로 Type 을 체크할 수 있다.
		var str = "Hello World";
		
		$.isString(str);		// true
		$.isNumber(str);		// false
		$.isObject(str);		// false
		$.isBoolean(str);		// false
		$.isFunction(str);		// false
		$.isArray(str);			// false
		$.isEmail(str);			// false
		$.isUrl(str);			// false
